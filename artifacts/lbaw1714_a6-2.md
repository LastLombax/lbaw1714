CREATE FUNCTION createInvoice() RETURNS TRIGGER AS
$BODY$
BEGIN
  IF NOT EXISTS (SELECT * 
    FROM "TicketType"
    WHERE "TicketType".idTicketType = New.type AND
    "TicketType".availableQuantity - $quantity < 0)
  RAISE EXCEPTION 'Not enough tickets avaiable of that type';
  END IF;

  UPDATE TicketType
  SET availableQuantity = aavailableQuantity - $quantity
  WHERE "TicketType".idTicketType = New.type;

  INSERT INTO "invoice" VALUES
  (id, memberid, taxpayer, address, $quantity, $quantity * price, $today)
  SELECT memberid, taxpayer, address
  FROM "member"
  WHERE (member.memberid = $_SESSION['currentMember'])
  SELECT price
  FROM "tickettype"
  WHERE (new.type = tickettype.idtickettype);
  RETURN NEW;
END
$BODY$
LANGUAGE plpgsql;

CREATE TRIGGER buyTicketBefore
  BEFORE INSERT ON "ticket"
  FOR EACH ROW
