```sql
CREATE FUNCTION inviteAllMembers() RETURNS TRIGGER AS
$BODY$
BEGIN
  /*
  se a comunidade do evento não for nula:
  - percorrer tabela de community member
  - para cada membro que seja membro da comunidade:
  INSERT INTO notification VALUES
  (id, timestamp, event(?), community, membro, contexto, null, evento);
  */
END
$BODY$
LANGUAGE plpgsql;

CREATE TRIGGER eventCommunity
  AFTER INSERT ON EVENT
  FOR EACH ROW
  WHEN Event.community NOT NULL
    EXECUTE PROCEDURE inviteAllMembers();
```
